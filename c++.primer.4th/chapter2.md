## Exercise 2.11

	int main() {
		int base, pow, result = 1;
		std::cout << "Enter base and exponent: ";
		std::cin >> base >> pow;
		for(int i = 0; i < pow; ++)

		std::cout << base
				<< " raised to the power of "
				<< pow << ": \t"
				<< result << std::endl;
	}
